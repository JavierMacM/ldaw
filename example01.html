En AngularJS el resultado de escribir 5+5 entre doble { } es: {{ 5 + 5 }}

AngularJs resuelve la expresion y regresa un resultado, escribe un color:
  • Red
  • Green
  • Yellow
  • Blue
  • Etc.
Aqui se hara uso de un controlador:

var app = angular.module("modulo1", []);
app.controller("controlador1", function($scope) {
$scope.nombre = "Javier";
$scope.appaterno = "Mendez";
$scope.apmaterno = "Martinez";
});

var app = angular.module("modulo2", []);
app.controller("controlador2", function($scope) {
$scope.nombre = "Juan";
$scope.appaterno = "Perez";
$scope.apmaterno = "Castro";
});

Al usar:
{ "Hola " + nombre + " " + appaterno + " " + apmaterno}

Se obtiene: {{ "Hola " + nombre + " " + appaterno + " " + apmaterno}}
